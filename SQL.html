Library System

            
				
			DROP SCHEMA IF EXISTS librarysytem;
			CREATE SCHEMA IF NOT EXISTS librarysytem;

			DROP TABLE IF EXISTS librarysytem.user_login;
			CREATE TABLE IF NOT EXISTS librarysytem.user_login (
				user_id TEXT PRIMARY KEY,
				user_password TEXT,
				first_name TEXT,
				last_name TEXT,
				sign_up_on DATE,
				email_id TEXT
			);

			DROP TABLE IF EXISTS librarysytem.publisher;
			CREATE TABLE IF NOT EXISTS librarysytem.publisher (
				publisher_id TEXT PRIMARY KEY,
				publisher TEXT,
				distributor TEXT,
				releases_count INT,
				last_release DATE
			);

			DROP TABLE IF EXISTS librarysytem.author;
			CREATE TABLE IF NOT EXISTS librarysytem.author (
				author_id TEXT PRIMARY KEY,
				first_name TEXT,
				last_name TEXT,
				publications_count INT
			);

			DROP TABLE IF EXISTS librarysytem.books;
			CREATE TABLE IF NOT EXISTS librarysytem.books (
				book_id TEXT PRIMARY KEY,
				book_code TEXT,
				book_name TEXT,
				author_id TEXT REFERENCES librarysytem.author (author_id),
				publisher_id TEXT REFERENCES librarysytem.publisher (publisher_id),
				book_version TEXT,
				release_date DATE,
				available_from DATE,
				is_available BOOLEAN
			);

			DROP TABLE IF EXISTS librarysytem.staff;
			CREATE TABLE IF NOT EXISTS librarysytem.staff (
				staff_id TEXT PRIMARY KEY,
				first_name TEXT,
				last_name TEXT,
				staff_role TEXT,
				start_date DATE,
				last_date DATE,
				is_active BOOLEAN,
				work_shift_start TIME,
				work_shift_end TIME
			);

			DROP TABLE IF EXISTS librarysytem.readers;
			CREATE TABLE IF NOT EXISTS librarysytem.readers (
				reader_id TEXT PRIMARY KEY,
				first_name TEXT,
				last_name TEXT,
				registered_on DATE,
				books_issued_total INT,
				books_issued_current INT,
				is_issued BOOLEAN,
				last_issue_date DATE,
				total_fine FLOAT,
				current_fine FLOAT
			);

			DROP TABLE IF EXISTS librarysytem.books_issue;
			CREATE TABLE IF NOT EXISTS librarysytem.books_issue (
				issue_id SERIAL PRIMARY KEY,
				book_id TEXT REFERENCES librarysytem.books (book_id),
				issued_to TEXT REFERENCES librarysytem.readers (reader_id),
				issued_on DATE,
				return_on DATE,
				current_fine FLOAT,
				fine_paid BOOLEAN,
				payment_transaction_id TEXT
			);

			DROP TABLE IF EXISTS librarysytem.settings;
			CREATE TABLE IF NOT EXISTS librarysytem.settings (
				book_issue_count_per_reader INT,
				fine_per_day FLOAT,
				book_return_in_days INT
			);

RAILWAY SYSTEM

            
				
				DROP SCHEMA IF EXISTS railsystem CASCADE;
				CREATE SCHEMA IF NOT EXISTS railsystem;
				
				DROP TABLE IF EXISTS railsystem.user_login;
				CREATE TABLE IF NOT EXISTS railsystem.user_login (
					user_id TEXT PRIMARY KEY,
					user_password TEXT,
					first_name TEXT,
					last_name TEXT,
					sign_up_on DATE,
					email_id TEXT
				);
				
				DROP TABLE IF EXISTS railsystem.passenger;
				CREATE TABLE IF NOT EXISTS railsystem.passenger (
					passenger_id TEXT PRIMARY KEY,
					user_password TEXT,
					first_name TEXT,
					last_name TEXT,
					sign_up_on DATE,
					email_id TEXT,
					contact TEXT
				);
				
				DROP TABLE IF EXISTS railsystem.train_type;
				CREATE TABLE IF NOT EXISTS railsystem.train_type (
					train_type_id TEXT PRIMARY KEY,
					train_type TEXT,
					coaches_count TEXT,
					passenger_strength TEXT,
					train_count DATE
				);
				
				DROP TABLE IF EXISTS railsystem.stations;
				CREATE TABLE IF NOT EXISTS railsystem.stations (
					station_id TEXT PRIMARY KEY,
					station_name TEXT,
					city TEXT,
					state TEXT
				);
				
				DROP TABLE IF EXISTS railsystem.train_details;
				CREATE TABLE IF NOT EXISTS railsystem.train_details (
					train_id TEXT PRIMARY KEY,
					train_type_id TEXT REFERENCES railsystem.train_type (train_type_id),
					source_station_id TEXT REFERENCES railsystem.stations (station_id),
					destination_station_id TEXT REFERENCES railsystem.stations (station_id),
					duration_minutes INT,
					journey_start TIMESTAMP,
					journey_end TIMESTAMP,
					passenger_strength INT,
					is_available BOOLEAN
				);
				
				DROP TABLE IF EXISTS railsystem.journey;
				CREATE TABLE IF NOT EXISTS railsystem.journey (
					journey_id TEXT PRIMARY KEY,
					passenger_id TEXT REFERENCES railsystem.passenger (passenger_id),
					train_id TEXT REFERENCES railsystem.train_details (train_id),
					booking_id TEXT,
					payment_id TEXT,
					payment_status TEXT,
					paid_on TIMESTAMP,
					booking_status TEXT,
					booked_on TIMESTAMP,
					seat_alloted TEXT,
					meal_booked BOOLEAN
				);
				
				DROP TABLE IF EXISTS railsystem.train_routes;
				CREATE TABLE IF NOT EXISTS railsystem.train_routes (
					row_id SERIAL PRIMARY KEY,
					route_id TEXT,
					train_id TEXT REFERENCES railsystem.train_details (train_id),
					station_id TEXT REFERENCES railsystem.stations (station_id),
					order_number INT,
					halt_duration_minutes INT,
					estimated_arrival TIME,
					estimated_departure TIME
				);

						recipe_id SERIAL PRIMARY KEY,
						name TEXT,
						description TEXT,
						instructions TEXT
					);
					
					DROP TABLE IF EXISTS readcook.ingredient;
					CREATE TABLE IF NOT EXISTS readcook.ingredient (
						ingredient_id SERIAL PRIMARY KEY,
						name TEXT
					);
					
					DROP TABLE IF EXISTS readcook.measure;
					CREATE TABLE IF NOT EXISTS readcook.measure (
						measure_id SERIAL PRIMARY KEY,
						name TEXT
					);
					
					DROP TABLE IF EXISTS readcook.recipe_ingredient;
					CREATE TABLE IF NOT EXISTS readcook.recipe_ingredient (
						recipe_id INT REFERENCES readcook.recipe (recipe_id),
						ingredient_id INT REFERENCES readcook.ingredient (ingredient_id),
						measure_id INT REFERENCES readcook.measure (measure_id),
						amount INT
					);
